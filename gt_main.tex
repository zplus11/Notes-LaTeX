\newcommand{\mysection}[1]{
	\begin{center}
		\begin{tikzpicture}[overlay]
			\fill[mycyan!90, draw = black] (0.025\linewidth, 0.5) rectangle (-.5\linewidth, -0.5);
			\fill[mypeach!90, draw = black] (0.025\linewidth, 0.5) rectangle (.5\linewidth, -0.5);
			\node[fill = white, draw = black, text = black] at (0, 0) {\LARGE\bfseries\sectionsfont #1};
		\end{tikzpicture}
	\end{center}
	\vspace{0.35cm}
}
\newcommand{\mysubsection}[1]{
	\begin{center}
		\begin{tikzpicture}[overlay]
			\draw[mypeach, line width = 10pt] (-.5\linewidth, 0) -- (0, 0);
			\draw[mycyan, line width = 10pt] (0, 0) -- (.5\linewidth, 0);
			\node[fill = white, text = black] at (0, 0) {\large\bfseries\sectionsfont #1};
		\end{tikzpicture}
	\end{center}
	\vspace{0.3cm}
}
